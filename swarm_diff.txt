+#!/usr/bin/env python
+import os
+import typer
+from datetime import datetime
+from langchain_openai import ChatOpenAI
+from langchain_community.checkpoints.sqlite import SqliteSaver
+from .graph import build_supervisor_graph
+from .nodes import PROJECT_RULES
+from typing import TypedDict
+
+app = typer.Typer()
+
+@ app.command()
+def main(task: str):
+    memory = SqliteSaver.from_conn_string("swarm_checkpoints.db")
+    supervisor_graph = build_supervisor_graph(memory)
+
+    config = {"configurable": {"thread_id": "cli-test"}}
+
+    input_state = {
+        "messages": [],
+        "current_task": task,
+        "subtasks": [],
+        "proposed_changes": "",
+        "review_feedback": "",
+        "final_decision": "",
+        "project_rules": PROJECT_RULES
+    }
+
+    result = supervisor_graph.invoke(input_state, config)
+
+    print("Final Result:")
+    print("Decision:", result.get("final_decision", "No decision"))
+    print("Feedback:", result.get("review_feedback", "No feedback"))
+    print("Changes:\\n", result.get("proposed_changes", "No changes"))
+    for msg in result["messages"]:
+        print(f"{msg['role']}: {msg['content']}")
+
+if __name__ == "__main__":
+    app()

+from langgraph.graph import StateGraph, END
+from langgraph.graph.message import add_messages
+from langchain_community.checkpoints.sqlite import SqliteSaver
+from .state import SupervisorState
+from .nodes import architect_node, code_writer_node, reviewer_node
+
+def build_supervisor_graph(checkpointer: SqliteSaver):
+    workflow = StateGraph(SupervisorState)
+
+    workflow.add_node("architect", architect_node)
+    workflow.add_node("code_writer", code_writer_node)
+    workflow.add_node("reviewer", reviewer_node)
+
+    workflow.set_entry_point("architect")
+    workflow.add_edge("architect", "code_writer")
+    workflow.add_edge("code_writer", "reviewer")
+
+    def should_continue(state: SupervisorState):
+        feedback = state.get("review_feedback", "").lower()
+        if "approve" in feedback or "approved" in feedback:
+            return "end"
+        return "architect"
+
+    workflow.add_conditional_edges(
+        "reviewer",
+        should_continue,
+        {
+            "architect": "architect",
+            "end": END
+        }
+    )
+
+    return workflow.compile(checkpointer=checkpointer)

+from langchain_openai import ChatOpenAI
+from langchain_core.prompts import ChatPromptTemplate
+import os
+from typing import Dict, Any
+
+PROJECT_RULES = {
+    "layered_packages": "Strict layered package structure under /packages/: core-llm, spiritual-knowledge, user-memory, agent-orchestration.",
+    "enneagram_support": "Always support Enneagram (type, wing, spiritual integration with ACIM).",
+    "acim_grounding": "Strong ACIM grounding in all responses.",
+    "do_not_remove": "Do NOT remove, weaken, or refactor anything related to Enneagram or ACIM lessons.",
+    "compatibility": "Preserve full compatibility with llama-cpp-server (OpenAI-compatible endpoint)."
+}
+
+llm = ChatOpenAI(
+    model='llama3.2',
+    base_url=os.getenv('LLM_BASE_URL', 'http://localhost:8080/v1'),
+    api_key='local',
+    temperature=0.1
+)
+
+def get_system_prompt(role: str) -> str:
+    rules_str = "\n".join([f"- {k}: {v}" for k,v in PROJECT_RULES.items()])
+    return f"""You are a {role} in kurs-bot development swarm. Follow these PROJECT RULES strictly:
+
+{rules_str}
+
+Keep responses concise, actionable. Output structured text."""
+
+architect_system = get_system_prompt("Software Architect")
+code_writer_system = get_system_prompt("Code Writer")
+reviewer_system = get_system_prompt("Code Reviewer")
+
+architect_prompt = ChatPromptTemplate.from_messages([
+    ("system", architect_system),
+    ("human", "Task: {current_task}\n\nPlan subtasks and next step.")
+])
+
+code_writer_prompt = ChatPromptTemplate.from_messages([
+    ("system", code_writer_system),
+    ("human", "Subtasks: {subtasks}\nCurrent: {current_task}\n\nGenerate proposed code changes as git diffs.")
+])
+
+reviewer_prompt = ChatPromptTemplate.from_messages([
+    ("system", reviewer_system),
+    ("human", "Proposed changes:\n{proposed_changes}\n\nReview. Output: APPROVE or NEEDS_REVISION\\nFeedback: ...")
+])
+
+def architect_node(state: Dict[str, Any]) -> Dict[str, Any]:
+    chain = architect_prompt | llm
+    result = chain.invoke(state)
+    content = result.content
+    subtasks = content.split('\n')[:3]  # dummy parse
+    return {
+        "messages": [{"role": "assistant", "content": content}],
+        "subtasks": subtasks,
+        "current_task": subtasks[0] if subtasks else state["current_task"]
+    }
+
+def code_writer_node(state: Dict[str, Any]) -> Dict[str, Any]:
+    chain = code_writer_prompt | llm
+    result = chain.invoke(state)
+    return {
+        "messages": [{"role": "assistant", "content": result.content}],
+        "proposed_changes": result.content
+    }
+
+def reviewer_node(state: Dict[str, Any]) -> Dict[str, Any]:
+    chain = reviewer_prompt | llm
+    result = chain.invoke(state)
+    feedback = result.content
+    decision = "APPROVE" if "APPROVE" in feedback.upper() else "NEEDS_REVISION"
+    return {
+        "messages": [{"role": "assistant", "content": feedback}],
+        "review_feedback": feedback,
+        "final_decision": decision
+    }
+
